from appwrite.client import Client
from appwrite.services.avatars import Avatars

client = Client()

client.set_project('<YOUR_PROJECT_ID>') # Your project ID

avatars = Avatars(client)

result = avatars.get_browser(

    width = 0, # optional
    height = 0, # optional
    quality = 0 # optional
)
