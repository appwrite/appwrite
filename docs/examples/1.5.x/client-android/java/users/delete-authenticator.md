    .setEndpoint("https://<REGION>.cloud.appwrite.io/v1") // Your API Endpoint

Users users = new Users(client);

    AuthenticatorProvider.TOTP,
    "[OTP]",

        Log.d("Appwrite", result.toString());
    })
